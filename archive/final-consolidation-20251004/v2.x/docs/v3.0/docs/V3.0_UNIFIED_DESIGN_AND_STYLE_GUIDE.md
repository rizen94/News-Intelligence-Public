```typescript
// ✅ v3.0: Composable architecture
// hooks/usePagination.ts
export const usePagination = (total: number, perPage: number = 12) => {
  const [page, setPage] = useState(1);
  const totalPages = Math.ceil(total / perPage);
  
  return {
    page,
    totalPages,
    hasNext: page < totalPages,
    hasPrev: page > 1,
    goToPage: setPage,
    nextPage: () => setPage(prev => Math.min(prev + 1, totalPages)),
    prevPage: () => setPage(prev => Math.max(prev - 1, 1))
  };
};

// hooks/useSearch.ts
export const useSearch = (initialQuery: string = '') => {
  const [query, setQuery] = useState(initialQuery);
  const [debouncedQuery, setDebouncedQuery] = useState(initialQuery);
  
  useEffect(() => {
    const timer = setTimeout(() => setDebouncedQuery(query), 300);
    return () => clearTimeout(timer);
  }, [query]);
  
  return { query, debouncedQuery, setQuery };
};

// Composed usage
const ArticlesPage = () => {
  const { query, setQuery } = useSearch();
  const { page, totalPages, goToPage } = usePagination(total);
  const { data: articles, loading, error } = useArticles({ query, page });
  
  // Composed from simple, reusable parts
};
```

### **9. 📈 SCALABILITY**
**"Design for growth from day one"**

**Code Style Requirements:**
- **Horizontal Scaling**: Stateless services that can scale horizontally
- **Database Scaling**: Proper indexing and query optimization
- **Caching Layers**: Multiple caching strategies for different data types
- **Microservices Ready**: Services designed to be independently deployable

**Implementation Standards:**
```typescript
// ✅ v3.0: Scalable architecture
// Stateless services
export class ArticlesService {
  constructor(
    private apiClient: APIClient,
    private cache: CacheService,
    private logger: Logger
  ) {}
  
  async getArticles(params: ArticleParams): Promise<PaginatedResponse<Article>> {
    // Stateless - can scale horizontally
    const cacheKey = `articles:${JSON.stringify(params)}`;
    
    const cached = await this.cache.get(cacheKey);
    if (cached) return cached;
    
    const result = await this.apiClient.getArticles(params);
    await this.cache.set(cacheKey, result, 300); // 5 minutes
    
    return result;
  }
}

// Database optimization
const getArticlesQuery = `
  SELECT id, title, content, source, published_date, created_at
  FROM articles 
  WHERE ($1::text IS NULL OR title ILIKE $1)
    AND ($2::text IS NULL OR source = $2)
  ORDER BY created_at DESC
  LIMIT $3 OFFSET $4
`;

// Proper indexing
CREATE INDEX CONCURRENTLY idx_articles_title_search ON articles 
USING gin(to_tsvector('english', title));

CREATE INDEX CONCURRENTLY idx_articles_source ON articles (source);
```
